    • зј–иҜ‘жқЎд»¶ v-if В  жҳҜжғ°жҖ§зҡ„пјҢеҰӮжһңеҲқе§ӢжқЎд»¶дёәеҒҮпјҢеҲҷд»Җд№Ҳд№ҹдёҚеҒҡгҖӮеҸӘжңүеңЁжқЎд»¶з¬¬дёҖж¬ЎеҸҳдёәзңҹж—¶жүҚејҖе§ӢеұҖйғЁзј–иҜ‘ v-show жҳҜеңЁд»»дҪ•жқЎд»¶дёӢ(йҰ–ж¬ЎжқЎд»¶жҳҜеҗҰдёәзңҹ)йғҪиў«зј–иҜ‘пјҢз„¶еҗҺиў«зј“еӯҳпјҢиҖҢдё” DOM е…ғзҙ дҝқз•ҷ
    • жҖ§иғҪж¶ҲиҖ— v-if В  жңүжӣҙй«ҳзҡ„еҲҮжҚўж¶ҲиҖ— v-show жңүжӣҙй«ҳзҡ„еҲқе§ӢжёІжҹ“ж¶ҲиҖ—
    • дҪҝз”ЁеңәжҷҜ v-if В  йҖӮеҗҲиҝҗиҗҘжқЎд»¶дёҚеӨ§еҸҜиғҪж”№еҸҳ v-show йҖӮеҗҲйў‘з№ҒеҲҮжҚў

еҰӮдҪ•и®© CSS еҖјеңЁеҪ“еүҚзҡ„з»„д»¶дёӯиө·дҪңз”Ё вҖӢ

еңЁ vue ж–Үд»¶дёӯзҡ„ style ж ҮзӯҫдёҠпјҢжңүдёҖдёӘзү№ж®Ҡзҡ„еұһжҖ§пјҡscopedгҖӮеҪ“дёҖдёӘ style ж ҮзӯҫжӢҘжңү scoped еұһжҖ§ж—¶пјҢе®ғзҡ„ CSS ж ·ејҸе°ұеҸӘиғҪдҪңз”ЁдәҺеҪ“еүҚзҡ„з»„д»¶пјҢд№ҹе°ұжҳҜиҜҙпјҢиҜҘж ·ејҸеҸӘиғҪйҖӮз”ЁдәҺеҪ“еүҚз»„д»¶е…ғзҙ гҖӮйҖҡиҝҮиҜҘеұһжҖ§пјҢеҸҜд»ҘдҪҝеҫ—з»„д»¶д№Ӣй—ҙзҡ„ж ·ејҸдёҚдә’зӣёжұЎжҹ“гҖӮеҰӮжһңдёҖдёӘйЎ№зӣ®дёӯзҡ„жүҖжңү style ж Үзӯҫе…ЁйғЁеҠ дёҠдәҶ scopedпјҢзӣёеҪ“дәҺе®һзҺ°дәҶж ·ејҸзҡ„жЁЎеқ—еҢ–гҖӮ

MVVMзҡ„дјҳзјәзӮ№? вҖӢ

дјҳзӮ№:

  • еҲҶзҰ»и§ҶеӣҫпјҲViewпјүе’ҢжЁЎеһӢпјҲModelпјүпјҢйҷҚдҪҺд»Јз ҒиҖҰеҗҲпјҢжҸҗвҫји§ҶеӣҫжҲ–иҖ…йҖ»иҫ‘зҡ„йҮҚвҪӨжҖ§: вҪҗеҰӮи§ҶеӣҫпјҲViewпјүеҸҜд»ҘзӢ¬вҪҙдәҺ Model еҸҳеҢ–е’Ңдҝ®ж”№пјҢвјҖдёӘ ViewModel еҸҜд»Ҙз»‘е®ҡдёҚеҗҢзҡ„"View"дёҠпјҢеҪ“ View еҸҳеҢ–зҡ„ж—¶еҖҷ Model дёҚеҸҜд»ҘдёҚеҸҳпјҢеҪ“ Model еҸҳеҢ–зҡ„ж—¶еҖҷ View д№ҹеҸҜд»ҘдёҚеҸҳгҖӮдҪ еҸҜд»ҘжҠҠвјҖдәӣи§ҶеӣҫйҖ»иҫ‘ж”ҫеңЁвјҖдёӘ ViewModel вҫҘвҫҜпјҢи®©еҫҲеӨҡ view йҮҚвҪӨиҝҷж®өи§ҶеӣҫйҖ»иҫ‘
  • жҸҗвҫјеҸҜжөӢиҜ•жҖ§: ViewModel зҡ„еӯҳеңЁеҸҜд»Ҙеё®еҠ©ејҖеҸ‘иҖ…жӣҙеҘҪең°зј–еҶҷжөӢиҜ•д»Јз Ғ
  • вҫғеҠЁжӣҙж–° dom: еҲ©вҪӨеҸҢеҗ‘з»‘е®ҡ,ж•°жҚ®жӣҙж–°еҗҺи§ҶеӣҫвҫғеҠЁжӣҙж–°,и®©ејҖеҸ‘иҖ…д»Һз№Ғзҗҗзҡ„вјҝеҠЁ dom дёӯи§Јж”ҫ

зјәзӮ№:

  • Bug еҫҲйҡҫиў«и°ғиҜ•: еӣ дёәдҪҝвҪӨеҸҢеҗ‘з»‘е®ҡзҡ„жЁЎејҸпјҢеҪ“дҪ зңӢеҲ°з•ҢвҫҜејӮеёёдәҶпјҢжңүеҸҜиғҪжҳҜдҪ  View зҡ„д»Јз Ғжңү BugпјҢд№ҹеҸҜиғҪжҳҜ Model зҡ„д»Јз Ғжңүй—®йўҳгҖӮж•°жҚ®з»‘е®ҡдҪҝеҫ—вјҖдёӘдҪҚзҪ®зҡ„ Bug иў«еҝ«йҖҹдј йҖ’еҲ°еҲ«зҡ„дҪҚзҪ®пјҢиҰҒе®ҡдҪҚеҺҹе§ӢеҮәй—®йўҳзҡ„ең°вҪ…е°ұеҸҳеҫ—дёҚйӮЈд№Ҳе®№жҳ“дәҶгҖӮеҸҰеӨ–пјҢж•°жҚ®з»‘е®ҡзҡ„еЈ°жҳҺжҳҜжҢҮд»ӨејҸең°еҶҷеңЁ View зҡ„жЁЎзүҲеҪ“дёӯзҡ„пјҢиҝҷдәӣеҶ…е®№жҳҜжІЎеҠһжі•еҺ»жү“ж–ӯзӮ№ debug зҡ„
  • вјҖдёӘвјӨзҡ„жЁЎеқ—дёӯ model д№ҹдјҡеҫҲвјӨпјҢиҷҪз„¶дҪҝвҪӨвҪ…дҫҝдәҶд№ҹеҫҲе®№жҳ“дҝқиҜҒдәҶж•°жҚ®зҡ„вјҖиҮҙжҖ§пјҢеҪ“ж—¶в»“жңҹжҢҒжңүпјҢдёҚйҮҠж”ҫеҶ…еӯҳе°ұйҖ жҲҗдәҶиҠұиҙ№жӣҙеӨҡзҡ„еҶ…еӯҳ
  • еҜ№дәҺвјӨеһӢзҡ„еӣҫеҪўеә”вҪӨзЁӢеәҸпјҢи§ҶеӣҫзҠ¶жҖҒиҫғеӨҡпјҢViewModel зҡ„жһ„е»әе’Ңз»ҙжҠӨзҡ„жҲҗжң¬йғҪдјҡвҪҗиҫғвҫјгҖӮ

MVVMгҖҒMVCгҖҒMVP зҡ„еҢәеҲ« вҖӢ

MVCгҖҒMVP е’Ң MVVM жҳҜдёүз§Қеёёи§Ғзҡ„иҪҜд»¶жһ¶жһ„и®ҫи®ЎжЁЎејҸпјҢдё»иҰҒйҖҡиҝҮеҲҶзҰ»е…іжіЁзӮ№зҡ„ж–№ејҸжқҘз»„з»Үд»Јз Ғз»“жһ„пјҢдјҳеҢ–ејҖеҸ‘ж•ҲзҺҮгҖӮ

еңЁејҖеҸ‘еҚ•йЎөйқўеә”з”Ёж—¶пјҢеҫҖеҫҖдёҖдёӘи·Ҝз”ұйЎөйқўеҜ№еә”дәҶдёҖдёӘи„ҡжң¬ж–Үд»¶пјҢжүҖжңүзҡ„йЎөйқўйҖ»иҫ‘йғҪеңЁдёҖдёӘи„ҡжң¬ж–Үд»¶йҮҢгҖӮйЎөйқўзҡ„жёІжҹ“гҖҒж•°жҚ®зҡ„иҺ·еҸ–пјҢеҜ№з”ЁжҲ·дәӢд»¶зҡ„е“Қеә”жүҖжңүзҡ„еә”з”ЁйҖ»иҫ‘йғҪж··еҗҲеңЁдёҖиө·пјҢиҝҷж ·еңЁејҖеҸ‘з®ҖеҚ•йЎ№зӣ®ж—¶пјҢеҸҜиғҪзңӢдёҚеҮәд»Җд№Ҳй—®йўҳпјҢеҰӮжһңйЎ№зӣ®еҸҳеҫ—еӨҚжқӮпјҢйӮЈд№Ҳж•ҙдёӘж–Үд»¶е°ұдјҡеҸҳеҫ—еҶ—й•ҝгҖҒж··д№ұпјҢиҝҷж ·еҜ№йЎ№зӣ®ејҖеҸ‘е’ҢеҗҺжңҹзҡ„йЎ№зӣ®з»ҙжҠӨжҳҜйқһеёёдёҚеҲ©зҡ„гҖӮ

пјҲ1пјүMVC

MVC йҖҡиҝҮеҲҶзҰ» ModelгҖҒView е’Ң Controller зҡ„ж–№ејҸжқҘз»„з»Үд»Јз Ғз»“жһ„гҖӮе…¶дёӯ View иҙҹиҙЈйЎөйқўзҡ„жҳҫзӨәйҖ»иҫ‘пјҢModel иҙҹиҙЈеӯҳеӮЁйЎөйқўзҡ„дёҡеҠЎж•°жҚ®пјҢд»ҘеҸҠеҜ№зӣёеә”ж•°жҚ®зҡ„ж“ҚдҪңгҖӮе№¶дё” View е’Ң Model еә”з”ЁдәҶи§ӮеҜҹиҖ…жЁЎејҸпјҢеҪ“ Model еұӮеҸ‘з”ҹж”№еҸҳзҡ„ж—¶еҖҷе®ғдјҡйҖҡзҹҘжңүе…і View еұӮжӣҙж–°йЎөйқўгҖӮController еұӮжҳҜ View еұӮе’Ң Model еұӮзҡ„зәҪеёҰпјҢе®ғдё»иҰҒиҙҹиҙЈз”ЁжҲ·дёҺеә”з”Ёзҡ„е“Қеә”ж“ҚдҪңпјҢеҪ“з”ЁжҲ·дёҺйЎөйқўдә§з”ҹдәӨдә’зҡ„ж—¶еҖҷпјҢController дёӯзҡ„дәӢд»¶и§ҰеҸ‘еҷЁе°ұејҖе§Ӣе·ҘдҪңдәҶпјҢйҖҡиҝҮи°ғз”Ё Model еұӮпјҢжқҘе®ҢжҲҗеҜ№ Model зҡ„дҝ®ж”№пјҢз„¶еҗҺ Model еұӮеҶҚеҺ»йҖҡзҹҘ View еұӮжӣҙж–°гҖӮ

пјҲ2пјүMVVM

MVVM еҲҶдёә ModelгҖҒViewгҖҒViewModelпјҡ

  • Model д»ЈиЎЁж•°жҚ®жЁЎеһӢпјҢж•°жҚ®е’ҢдёҡеҠЎйҖ»иҫ‘йғҪеңЁ Model еұӮдёӯе®ҡд№үпјӣ
  • View д»ЈиЎЁ UI и§ҶеӣҫпјҢиҙҹиҙЈж•°жҚ®зҡ„еұ•зӨәпјӣ
  • ViewModel иҙҹиҙЈзӣ‘еҗ¬ Model дёӯж•°жҚ®зҡ„ж”№еҸҳе№¶дё”жҺ§еҲ¶и§Ҷеӣҫзҡ„жӣҙж–°пјҢеӨ„зҗҶз”ЁжҲ·дәӨдә’ж“ҚдҪңпјӣ

Model е’Ң View е№¶ж— зӣҙжҺҘе…іиҒ”пјҢиҖҢжҳҜйҖҡиҝҮ ViewModel жқҘиҝӣиЎҢиҒ”зі»зҡ„пјҢModel е’Ң ViewModel д№Ӣй—ҙжңүзқҖеҸҢеҗ‘ж•°жҚ®з»‘е®ҡзҡ„иҒ”зі»гҖӮеӣ жӯӨеҪ“ Model дёӯзҡ„ж•°жҚ®ж”№еҸҳж—¶дјҡи§ҰеҸ‘ View еұӮзҡ„еҲ·ж–°пјҢView дёӯз”ұдәҺз”ЁжҲ·дәӨдә’ж“ҚдҪңиҖҢж”№еҸҳзҡ„ж•°жҚ®д№ҹдјҡеңЁ Model дёӯеҗҢжӯҘгҖӮ

иҝҷз§ҚжЁЎејҸе®һзҺ°дәҶ Model е’Ң View зҡ„ж•°жҚ®иҮӘеҠЁеҗҢжӯҘпјҢеӣ жӯӨејҖеҸ‘иҖ…еҸӘйңҖиҰҒдё“жіЁдәҺж•°жҚ®зҡ„з»ҙжҠӨж“ҚдҪңеҚіеҸҜпјҢиҖҢдёҚйңҖиҰҒиҮӘе·ұж“ҚдҪң DOMгҖӮ

пјҲ3пјүMVP

MVP жЁЎејҸдёҺ MVC е”ҜдёҖдёҚеҗҢзҡ„еңЁдәҺ Presenter е’Ң ControllerгҖӮеңЁ MVC жЁЎејҸдёӯдҪҝз”Ёи§ӮеҜҹиҖ…жЁЎејҸпјҢжқҘе®һзҺ°еҪ“ Model еұӮж•°жҚ®еҸ‘з”ҹеҸҳеҢ–зҡ„ж—¶еҖҷпјҢйҖҡзҹҘ View еұӮзҡ„жӣҙж–°гҖӮиҝҷж · View еұӮе’Ң Model еұӮиҖҰеҗҲеңЁдёҖиө·пјҢеҪ“йЎ№зӣ®йҖ»иҫ‘еҸҳеҫ—еӨҚжқӮзҡ„ж—¶еҖҷпјҢеҸҜиғҪдјҡйҖ жҲҗд»Јз Ғзҡ„ж··д№ұпјҢе№¶дё”еҸҜиғҪдјҡеҜ№д»Јз Ғзҡ„еӨҚз”ЁжҖ§йҖ жҲҗдёҖдәӣй—®йўҳгҖӮMVP зҡ„жЁЎејҸйҖҡиҝҮдҪҝз”Ё Presenter жқҘе®һзҺ°еҜ№ View еұӮе’Ң Model еұӮзҡ„и§ЈиҖҰгҖӮMVC дёӯзҡ„ Controller еҸӘзҹҘйҒ“ Model зҡ„жҺҘеҸЈпјҢеӣ жӯӨе®ғжІЎжңүеҠһжі•жҺ§еҲ¶ View еұӮзҡ„жӣҙж–°пјҢMVP жЁЎејҸдёӯпјҢView еұӮзҡ„жҺҘеҸЈжҡҙйңІз»ҷдәҶ Presenter еӣ жӯӨеҸҜд»ҘеңЁ Presenter дёӯе°Ҷ Model зҡ„еҸҳеҢ–е’Ң View зҡ„еҸҳеҢ–з»‘е®ҡеңЁдёҖиө·пјҢд»ҘжӯӨжқҘе®һзҺ° View е’Ң Model зҡ„еҗҢжӯҘжӣҙж–°гҖӮиҝҷж ·е°ұе®һзҺ°дәҶеҜ№ View е’Ң Model зҡ„и§ЈиҖҰпјҢPresenter иҝҳеҢ…еҗ«дәҶе…¶д»–зҡ„е“Қеә”йҖ»иҫ‘гҖӮ

и°ҲдёҖи°ҲеҜ№ MVVM зҡ„зҗҶи§Јпјҹ вҖӢ

  • MVVM жҳҜ Model-View-ViewModel зҡ„зј©еҶҷгҖӮMVVM жҳҜдёҖз§Қи®ҫи®ЎжҖқжғігҖӮ
  • Model еұӮд»ЈиЎЁж•°жҚ®жЁЎеһӢпјҢд№ҹеҸҜд»ҘеңЁ Model дёӯе®ҡд№үж•°жҚ®дҝ®ж”№е’Ңж“ҚдҪңзҡ„дёҡеҠЎйҖ»иҫ‘;
  • View д»ЈиЎЁ UI з»„д»¶пјҢе®ғиҙҹиҙЈе°Ҷж•°жҚ®жЁЎеһӢиҪ¬еҢ–жҲҗ UI еұ•зҺ°еҮәжқҘпјҢView жҳҜдёҖдёӘеҗҢжӯҘ View е’Ң Model зҡ„еҜ№иұЎ

push

Vue зҡ„е“Қеә”ејҸзі»з»ҹеҲҷжҳҜ push зҡ„д»ЈиЎЁпјҢеҪ“ Vue зЁӢеәҸеҲқе§ӢеҢ–зҡ„ж—¶еҖҷе°ұдјҡеҜ№ж•°жҚ® data иҝӣиЎҢдҫқиө–зҡ„ж”¶йӣҶпјҢдёҖдҪҶж•°жҚ®еҸ‘з”ҹеҸҳеҢ–пјҢе“Қеә”ејҸзі»з»ҹе°ұдјҡз«ӢеҲ»еҫ—зҹҘпјҢеӣ жӯӨ Vue жҳҜдёҖејҖе§Ӣе°ұзҹҘйҒ“жҳҜгҖҢеңЁе“ӘеҸ‘з”ҹеҸҳеҢ–дәҶгҖҚ

дҪҶжҳҜиҝҷеҸҲдјҡдә§з”ҹдёҖдёӘй—®йўҳпјҢйҖҡеёёз»‘е®ҡдёҖдёӘж•°жҚ®е°ұйңҖиҰҒдёҖдёӘ WatcherпјҢдёҖдҪҶжҲ‘д»¬зҡ„з»‘е®ҡз»ҶзІ’еәҰиҝҮй«ҳе°ұдјҡдә§з”ҹеӨ§йҮҸзҡ„ WatcherпјҢиҝҷдјҡеёҰжқҘеҶ…еӯҳд»ҘеҸҠдҫқиө–иҝҪиёӘзҡ„ејҖй”ҖпјҢиҖҢз»ҶзІ’еәҰиҝҮдҪҺдјҡж— жі•зІҫеҮҶдҫҰжөӢеҸҳеҢ–пјҢеӣ жӯӨ Vue зҡ„и®ҫи®ЎжҳҜйҖүжӢ©дёӯзӯүз»ҶзІ’еәҰзҡ„ж–№жЎҲпјҢеңЁз»„д»¶зә§еҲ«иҝӣиЎҢ push дҫҰжөӢзҡ„ж–№ејҸпјҢд№ҹе°ұжҳҜйӮЈеҘ—е“Қеә”ејҸзі»з»ҹгҖӮ

йҖҡеёёжҲ‘д»¬дјҡз¬¬дёҖж—¶й—ҙдҫҰжөӢеҲ°еҸ‘з”ҹеҸҳеҢ–зҡ„з»„д»¶,з„¶еҗҺеңЁз»„д»¶еҶ…йғЁиҝӣиЎҢ Virtual Dom Diff иҺ·еҸ–жӣҙеҠ е…·дҪ“зҡ„е·®ејӮпјҢиҖҢ Virtual Dom Diff еҲҷжҳҜ pull ж“ҚдҪңпјҢVue жҳҜ push + pull з»“еҗҲзҡ„ж–№ејҸиҝӣиЎҢеҸҳеҢ–дҫҰжөӢзҡ„гҖӮ

Vue дёәд»Җд№ҲжІЎжңүзұ»дјјдәҺ React дёӯ shouldComponentUpdate зҡ„з”ҹе‘Ҫе‘Ёжңҹпјҹ вҖӢ

ж №жң¬еҺҹеӣ жҳҜ Vue дёҺ React зҡ„еҸҳеҢ–дҫҰжөӢж–№ејҸжңүжүҖдёҚеҗҢ

React жҳҜ pull зҡ„ж–№ејҸдҫҰжөӢеҸҳеҢ–пјҢеҪ“ React зҹҘйҒ“еҸ‘з”ҹеҸҳеҢ–еҗҺпјҢдјҡдҪҝз”Ё Virtual Dom Diff иҝӣиЎҢе·®ејӮжЈҖжөӢ,дҪҶжҳҜеҫҲеӨҡз»„д»¶е®һйҷ…дёҠжҳҜиӮҜе®ҡдёҚдјҡеҸ‘з”ҹеҸҳеҢ–зҡ„пјҢиҝҷдёӘж—¶еҖҷйңҖиҰҒз”Ё shouldComponentUpdate иҝӣиЎҢжүӢеҠЁж“ҚдҪңжқҘеҮҸе°‘ diffпјҢд»ҺиҖҢжҸҗй«ҳзЁӢеәҸж•ҙдҪ“зҡ„жҖ§иғҪгҖӮ

Vue жҳҜ pull+push зҡ„ж–№ејҸдҫҰжөӢеҸҳеҢ–зҡ„пјҢеңЁдёҖејҖе§Ӣе°ұзҹҘйҒ“йӮЈдёӘз»„д»¶еҸ‘з”ҹдәҶеҸҳеҢ–пјҢеӣ жӯӨеңЁ push зҡ„йҳ¶ж®өе№¶дёҚйңҖиҰҒжүӢеҠЁжҺ§еҲ¶ diffпјҢиҖҢз»„д»¶еҶ…йғЁйҮҮз”Ёзҡ„ diff ж–№ејҸе®һйҷ…дёҠжҳҜеҸҜд»Ҙеј•е…Ҙзұ»дјјдәҺ shouldComponentUpdate зӣёе…із”ҹе‘Ҫе‘Ёжңҹзҡ„пјҢдҪҶжҳҜйҖҡеёёеҗҲзҗҶеӨ§е°Ҹзҡ„з»„д»¶дёҚдјҡжңүиҝҮйҮҸзҡ„ diffпјҢжүӢеҠЁдјҳеҢ–зҡ„д»·еҖјжңүйҷҗпјҢеӣ жӯӨзӣ®еүҚ Vue е№¶жІЎжңүиҖғиҷ‘еј•е…Ҙ shouldComponentUpdate иҝҷз§ҚжүӢеҠЁдјҳеҢ–зҡ„з”ҹе‘Ҫе‘ЁжңҹгҖӮ

еҺҹз”ҹзҡ„дәӢд»¶з»‘е®ҡеңЁжҷ®йҖҡе…ғзҙ дёҠжҳҜйҖҡиҝҮ @click _ иҝӣиЎҢз»‘е®ҡпјҢеңЁз»„д»¶дёҠжҳҜйҖҡиҝҮ @click.native иҝӣиЎҢз»‘е®ҡпјҢз»„д»¶дёӯзҡ„ _nativeOn_ жҳҜзӯүд»·дәҺ on зҡ„гҖӮз»„д»¶зҡ„дәӢд»¶з»‘е®ҡзҡ„ @click жҳҜ vue дёӯиҮӘе®ҡд№үзҡ„ $on ж–№жі•жқҘе®һзҺ°зҡ„пјҢеҝ…йЎ»жңү $emit жүҚеҸҜд»Ҙи§ҰеҸ‘гҖӮ

еҺҹз”ҹдәӢд»¶з»‘е®ҡеҺҹзҗҶ

еңЁ runtime дёӢзҡ„ patch.js дёӯ createPatchFunction жү§иЎҢдәҶд№ӢеҗҺеҶҚиөӢеҖјз»ҷ patchгҖӮ

createPatchFunction ж–№жі•жңүдёӨдёӘеҸӮж•°пјҢеҲҶеҲ«жҳҜ nodeOps еӯҳж”ҫж“ҚдҪң dom иҠӮзӮ№зҡ„ж–№жі•е’Ң modulesпјҢmodules жҳҜжңүдёӨдёӘж•°з»„жӢјжҺҘиө·жқҘзҡ„пјҢmodules жӢјжҺҘе®Ңзҡ„ж•°з»„дёӯжңүдёҖдёӘе…ғзҙ е°ұжҳҜ eventsпјҢдәӢд»¶ж·»еҠ е°ұеҸ‘з”ҹеңЁиҝҷйҮҢгҖӮ

events е…ғзҙ е…іиҒ”зҡ„е°ұжҳҜ events.js ж–Үд»¶пјҢеңЁ events дёӯжңүдёҖдёӘ updateDOMListeners ж–№жі•пјҢеңЁ events ж–Үд»¶зҡ„з»“е°ҫеҜјеҮәдәҶдёҖдёӘеҜ№иұЎпјҢз„¶еҗҺеҜ№иұЎжңүдёҖдёӘеұһжҖ§еҸ«еҒҡ createпјҢиҝҷдёӘеұһжҖ§е…іиҒ”зҡ„е°ұжҳҜ updateDOMListeners ж–№жі•гҖӮ

еңЁжү§иЎҢ createPatchFunction ж–№жі•ж—¶пјҢе°ұдјҡе°ҶиҝҷдёӨдёӘеҸӮж•°дј е…ҘпјҢеңЁ createPatchFunction ж–№жі•дёӯжҺҘж”¶дәҶдёҖдёӘеҸӮж•° backendпјҢеңЁиҜҘж–№жі•дёӯдёҖејҖе§ӢиҝӣиЎҢ backend зҡ„и§Јжһ„пјҢе°ұжҳҜдёҠйқўзҡ„ nodeOps е’Ң modules еҸӮж•°пјҢи§Јжһ„е®Ңд№ӢеҗҺиҝӣе…Ҙ for еҫӘзҺҜгҖӮ

еңЁ createPatchFunction ејҖеӨҙе®ҡд№үдәҶдёҖдёӘ cbs еҜ№иұЎгҖӮfor еҫӘзҺҜйҒҚеҺҶдёҖдёӘеҸ« hooks зҡ„ж•°з»„гҖӮhooks жҳҜж–Үд»¶дёҖејҖеӨҙе®ҡд№үзҡ„дёҖдёӘж•°з»„пјҢе…¶дёӯеҢ…жӢ¬жңү createпјҢfor еҫӘзҺҜе°ұжҳҜеңЁ cbs дёҠе®ҡд№үдёҖзі»еҲ—е’Ң hooks е…ғзҙ зӣёеҗҢзҡ„еұһжҖ§пјҢз„¶еҗҺй”®еҖјжҳҜдёҖдёӘж•°з»„пјҢз„¶еҗҺж•°з»„еҶ…е®№жҳҜ modules йҮҢйқўзҡ„дёҖдәӣеҶ…е®№гҖӮиҝҷж—¶е°ұжҠҠ events ж–Үд»¶дёӯеҜјеҮәжқҘзҡ„ create еұһжҖ§ж”ҫеңЁдәҶ cbs дёҠгҖӮ

еҪ“жҲ‘д»¬иҝӣе…ҘйҰ–ж¬ЎжёІжҹ“зҡ„ж—¶еҖҷпјҢдјҡжү§иЎҢеҲ° patch еҮҪж•°йҮҢйқўзҡ„ createElm ж–№жі•пјҢиҝҷдёӘж–№жі•дёӯе°ұдјҡи°ғз”Ё invokeCreateHooks еҮҪж•°пјҢз”ЁжқҘеӨ„зҗҶдәӢд»¶зі»з»ҹпјҢиҝҷйҮҢе°ұжҳҜзңҹжӯЈеҮҶеӨҮиҝӣиЎҢеҺҹз”ҹдәӢд»¶з»‘е®ҡзҡ„е…ҘеҸЈгҖӮinvokeCreateHooks ж–№жі•дёӯпјҢйҒҚеҺҶдәҶ cbs.create ж•°з»„йҮҢйқўзҡ„еҶ…е®№гҖӮз„¶еҗҺжҠҠ cbs.create йҮҢйқўзҡ„еҮҪж•°е…ЁйғЁйғҪжү§иЎҢдёҖж¬ЎпјҢеңЁ cbs.create е…¶дёӯдёҖдёӘеҮҪж•°е°ұжҳҜ updateDOMListenersгҖӮ

v-if иғҪеӨҹжҺ§еҲ¶жҳҜеҗҰз”ҹжҲҗ vnodeпјҢд№ҹе°ұй—ҙжҺҘжҺ§еҲ¶дәҶжҳҜеҗҰз”ҹжҲҗеҜ№еә”зҡ„ domгҖӮеҪ“ v-if дёә true ж—¶пјҢдјҡз”ҹжҲҗеҜ№еә”зҡ„ vnodeпјҢе№¶з”ҹжҲҗеҜ№еә”зҡ„ dom е…ғзҙ пјӣеҪ“е…¶дёә false ж—¶пјҢдёҚдјҡз”ҹжҲҗеҜ№еә”зҡ„ vnodeпјҢиҮӘз„¶дёҚдјҡз”ҹжҲҗд»»дҪ•зҡ„ dom е…ғзҙ гҖӮ v-show е§Ӣз»Ҳдјҡз”ҹжҲҗ vnodeпјҢд№ҹе°ұй—ҙжҺҘеҜјиҮҙдәҶе§Ӣз»Ҳз”ҹжҲҗ domгҖӮе®ғеҸӘжҳҜжҺ§еҲ¶ dom зҡ„ display еұһжҖ§пјҢеҪ“ v-show дёә true ж—¶пјҢдёҚеҒҡд»»дҪ•еӨ„зҗҶпјӣеҪ“е…¶дёә false ж—¶пјҢз”ҹжҲҗзҡ„ dom зҡ„ display еұһжҖ§дёә noneгҖӮ дҪҝз”Ё v-if еҸҜд»Ҙжңүж•Ҳзҡ„еҮҸе°‘ж ‘зҡ„иҠӮзӮ№е’ҢжёІжҹ“йҮҸпјҢдҪҶд№ҹдјҡеҜјиҮҙж ‘зҡ„дёҚзЁіе®ҡпјӣиҖҢдҪҝз”Ё v-show еҸҜд»ҘдҝқжҢҒж ‘зҡ„зЁіе®ҡпјҢдҪҶдёҚиғҪеҮҸе°‘ж ‘зҡ„иҠӮзӮ№е’ҢжёІжҹ“йҮҸгҖӮ еӣ жӯӨпјҢеңЁе®һйҷ…ејҖеҸ‘дёӯпјҢжҳҫзӨәзҠ¶жҖҒеҸҳеҢ–йў‘з№Ғзҡ„жғ…еҶөдёӢеә”иҜҘдҪҝз”Ё v-showпјҢд»ҘдҝқжҢҒж ‘зҡ„зЁіе®ҡпјӣжҳҫзӨәзҠ¶жҖҒеҸҳеҢ–иҫғе°‘ж—¶еә”иҜҘдҪҝз”Ё v-ifпјҢд»ҘеҮҸе°‘ж ‘зҡ„иҠӮзӮ№е’ҢжёІжҹ“йҮҸгҖӮ

data дёәд»Җд№ҲжҳҜдёҖдёӘеҮҪж•°иҖҢдёҚжҳҜеҜ№иұЎ вҖӢ

JavaScript дёӯзҡ„еҜ№иұЎжҳҜеј•з”Ёзұ»еһӢзҡ„ж•°жҚ®пјҢеҪ“еӨҡдёӘе®һдҫӢеј•з”ЁеҗҢдёҖдёӘеҜ№иұЎж—¶пјҢеҸӘиҰҒдёҖдёӘе®һдҫӢеҜ№иҝҷдёӘеҜ№иұЎиҝӣиЎҢж“ҚдҪңпјҢе…¶д»–е®һдҫӢдёӯзҡ„ж•°жҚ®д№ҹдјҡеҸ‘з”ҹеҸҳеҢ–гҖӮ иҖҢеңЁ Vue дёӯпјҢжӣҙеӨҡзҡ„жҳҜжғіиҰҒеӨҚз”Ёз»„д»¶пјҢйӮЈе°ұйңҖиҰҒжҜҸдёӘз»„д»¶йғҪжңүиҮӘе·ұзҡ„ж•°жҚ®пјҢиҝҷж ·з»„д»¶д№Ӣй—ҙжүҚдёҚдјҡзӣёдә’е№Іжү°гҖӮ жүҖд»Ҙз»„д»¶зҡ„ж•°жҚ®дёҚиғҪеҶҷжҲҗеҜ№иұЎзҡ„еҪўејҸпјҢиҖҢжҳҜиҰҒеҶҷжҲҗеҮҪж•°зҡ„еҪўејҸгҖӮж•°жҚ®д»ҘеҮҪж•°иҝ”еӣһеҖјзҡ„еҪўејҸе®ҡд№үпјҢиҝҷж ·еҪ“жҜҸж¬ЎеӨҚз”Ёз»„д»¶зҡ„ж—¶еҖҷпјҢе°ұдјҡиҝ”еӣһдёҖдёӘж–°зҡ„ dataпјҢд№ҹе°ұжҳҜиҜҙжҜҸдёӘз»„д»¶йғҪжңүиҮӘе·ұзҡ„з§Ғжңүж•°жҚ®з©әй—ҙпјҢе®ғд»¬еҗ„иҮӘз»ҙжҠӨиҮӘе·ұзҡ„ж•°жҚ®пјҢдёҚдјҡе№Іжү°е…¶д»–з»„д»¶зҡ„жӯЈеёёиҝҗиЎҢгҖӮ

Vue дёӯе°ҒиЈ…зҡ„ж•°з»„ж–№жі•жңүе“ӘдәӣпјҢе…¶еҰӮдҪ•е®һзҺ°йЎөйқўжӣҙж–° вҖӢ

еңЁ Vue дёӯпјҢеҜ№е“Қеә”ејҸеӨ„зҗҶеҲ©з”Ёзҡ„жҳҜ Object.defineProperty еҜ№ж•°жҚ®иҝӣиЎҢжӢҰжҲӘпјҢиҖҢиҝҷдёӘж–№жі•е№¶дёҚиғҪзӣ‘еҗ¬еҲ°ж•°з»„еҶ…йғЁеҸҳеҢ–пјҢж•°з»„й•ҝеәҰеҸҳеҢ–пјҢж•°з»„зҡ„жҲӘеҸ–еҸҳеҢ–зӯүпјҢжүҖд»ҘйңҖиҰҒеҜ№иҝҷдәӣж“ҚдҪңиҝӣиЎҢ hackпјҢи®© Vue иғҪзӣ‘еҗ¬еҲ°е…¶дёӯзҡ„еҸҳеҢ–гҖӮ push() pop() shift() unshift () splice() sort() reverse() йӮЈ Vue жҳҜеҰӮдҪ•е®һзҺ°и®©иҝҷдәӣж•°з»„ж–№жі•е®һзҺ°е…ғзҙ зҡ„е®һж—¶жӣҙж–°зҡ„е‘ўпјҢдёӢйқўжҳҜ Vue дёӯеҜ№иҝҷдәӣж–№жі•зҡ„е°ҒиЈ…пјҡ

javascript
// зј“еӯҳж•°з»„еҺҹеһӢ
const arrayProto = Array.prototype;
// е®һзҺ° arrayMethods.__proto__ === Array.prototype
export const arrayMethods = Object.create(arrayProto);
// йңҖиҰҒиҝӣиЎҢеҠҹиғҪжӢ“еұ•зҡ„ж–№жі•
const methodsToPatch = [
  "push",
  "pop",
  "shift",
  "unshift",
  "splice",
  "sort",
  "reverse",
];

/**
 * Intercept mutating methods and emit events
 */
methodsToPatch.forEach(function (method) {
  // зј“еӯҳеҺҹз”ҹж•°з»„ж–№жі•
  const original = arrayProto[method];
  def(arrayMethods, method, function mutator(...args) {
    // жү§иЎҢе№¶зј“еӯҳеҺҹз”ҹж•°з»„еҠҹиғҪ
    const result = original.apply(this, args);
    // е“Қеә”ејҸеӨ„зҗҶ
    const ob = this.__ob__;
    let inserted;
    switch (method) {
      // pushгҖҒunshiftдјҡж–°еўһзҙўеј•пјҢжүҖд»ҘиҰҒжүӢеҠЁobserver
      case "push":
      case "unshift":
        inserted = args;
        break;
      // spliceж–№жі•пјҢеҰӮжһңдј е…ҘдәҶз¬¬дёүдёӘеҸӮж•°пјҢд№ҹдјҡжңүзҙўеј•еҠ е…ҘпјҢд№ҹиҰҒжүӢеҠЁobserverгҖӮ
      case "splice":
        inserted = args.slice(2);
        break;
    }
    //
    if (inserted) ob.observeArray(inserted); // иҺ·еҸ–жҸ’е…Ҙзҡ„еҖјпјҢе№¶и®ҫзҪ®е“Қеә”ејҸзӣ‘еҗ¬
    // notify change
    ob.dep.notify(); // йҖҡзҹҘдҫқиө–жӣҙж–°
    // иҝ”еӣһеҺҹз”ҹж•°з»„ж–№жі•зҡ„жү§иЎҢз»“жһң
    return result;
  });
});

иҷҪз„¶ж–ҮжЎЈдёҚе»әи®®еңЁеә”з”ЁдёӯзӣҙжҺҘдҪҝз”Ё mixinпјҢдҪҶжҳҜеҰӮжһңдёҚж»Ҙз”Ёзҡ„иҜқд№ҹжҳҜеҫҲжңүеё®еҠ©зҡ„пјҢжҜ”еҰӮеҸҜд»Ҙе…ЁеұҖж··е…Ҙе°ҒиЈ…еҘҪзҡ„ ajax жҲ–иҖ…дёҖдәӣе·Ҙе…·еҮҪж•°зӯүзӯүгҖӮ

mixins еә”иҜҘжҳҜжңҖеёёдҪҝз”Ёзҡ„жү©еұ•з»„д»¶зҡ„ж–№ејҸдәҶгҖӮеҰӮжһңеӨҡдёӘз»„д»¶дёӯжңүзӣёеҗҢзҡ„дёҡеҠЎйҖ»иҫ‘пјҢе°ұеҸҜд»Ҙе°ҶиҝҷдәӣйҖ»иҫ‘еүҘзҰ»еҮәжқҘпјҢйҖҡиҝҮ mixins ж··е…Ҙд»Јз ҒпјҢжҜ”еҰӮдёҠжӢүдёӢжӢүеҠ иҪҪж•°жҚ®иҝҷз§ҚйҖ»иҫ‘зӯүзӯүгҖӮ еҸҰеӨ–йңҖиҰҒжіЁж„Ҹзҡ„жҳҜ mixins ж··е…Ҙзҡ„й’©еӯҗеҮҪж•°дјҡе…ҲдәҺз»„д»¶еҶ…зҡ„й’©еӯҗеҮҪж•°жү§иЎҢпјҢе№¶дё”еңЁйҒҮеҲ°еҗҢеҗҚйҖүйЎ№зҡ„ж—¶еҖҷд№ҹдјҡжңүйҖүжӢ©жҖ§зҡ„иҝӣиЎҢеҗҲе№¶гҖӮ

ж—¶жңә

Transitionз»„д»¶дјҡзӣ‘жҺ§slotдёӯе”ҜдёҖж №е…ғзҙ зҡ„еҮәзҺ°е’Ңж¶ҲеӨұпјҢе№¶дјҡеңЁе…¶еҮәзҺ°е’Ңж¶ҲеӨұж—¶еә”з”ЁиҝҮжёЎж•Ҳжһң TransitionдёҚз”ҹжҲҗд»»дҪ•е…ғзҙ пјҢеҸӘжҳҜдёәдәҶз”ҹжҲҗиҝҮжёЎж•Ҳжһң е…·дҪ“зҡ„зӣ‘еҗ¬еҶ…е®№жҳҜпјҡ

  • е®ғдјҡеҜ№ж–°ж—§дёӨдёӘиҷҡжӢҹиҠӮзӮ№иҝӣиЎҢеҜ№жҜ”пјҢеҰӮжһңж—§иҠӮзӮ№иў«й”ҖжҜҒпјҢеҲҷеә”з”Ёж¶ҲеӨұж•ҲжһңпјҢеҰӮжһңж–°иҠӮзӮ№жҳҜж–°еўһзҡ„пјҢеҲҷеә”з”Ёиҝӣе…Ҙж•Ҳжһң
  • еҰӮжһңдёҚжҳҜдёҠиҝ°жғ…еҶөпјҢеҲҷе®ғдјҡеҜ№жҜ”ж–°ж—§иҠӮзӮ№пјҢи§ӮеҜҹе…¶v-showжҳҜеҗҰеҸҳеҢ–пјҢtrue->falseеә”з”Ёж¶ҲеӨұж•ҲжһңпјҢfalse->trueеә”з”Ёиҝӣе…Ҙж•Ҳжһң
